`Eles podem ser visto como uma versão de comportamento e atributos do objeto para a classe primitiva`


```Java

//byte

Byte b = 100;

Short s = 1000;

Integer i = Integer.parseInt("10000");

//Inicilizando assim a classe é possivel converter para String

Long l = 10000L;

System.out.println(b.byteValue());

//Exatamente o valor byte que esta nessa classe

System.out.println(s.toString());

//Transforma o valor em String

System.out.println(i);

System.out.println(l);

Float f = 123.10F;

System.out.println(f);

Double d = 1234.5678;

System.out.println(d);

Boolean bo = Boolean.parseBoolean("true");

//A gente passouo valor booleano para uma string... sim, isso é possivel

Character c = '#';//char

System.out.println(c.toString());

}

}
```
