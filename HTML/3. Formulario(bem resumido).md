                <option value="adulto">Entre 18 e 60</option>

        <input type="radio" name="sexo" value="f"/> Feminino
        <br />
        Redes Sociais:
        <br />
        <input type="checkbox" name="redes" value="face"/> Facebook
        <input type="checkbox" name="redes" value="google"/> Google
        <br />
        <button type="submit">Enviar</button>
        <button type="reset">Limpar</button>

</form>    

</body>
</html>
